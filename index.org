** Hello, Go with fmt
- But it's more idiomatic to import fmt
- Also, fmt.Println is more flexible than println

#+BEGIN_SRC go
package main

import "fmt"

func main() {
	fmt.Println("Hello, World!")
}
#+END_SRC

* Hello, Elixir!

#+BEGIN_SRC elixir
#!/usr/bin/env elixir

greet = fn s -> IO.puts "Hello, #{s}!" end

if length(System.argv) == 0 do
  greet.("World")
else 
  Enum.each System.argv, greet
end
#+END_SRC
